JavaSE基础语法

2022/12/29 学习 Java

String是引用类
　　　本来需要new来申请存储空间
　　　　String x = new String(“*”);
　　　　String y = “&”;
　　　String 用双引号 “absolute”，里面是字符串
　　　char 用单引号 ‘b’，里面是字符

break和continue
　　　break：跳出循环，整个循环结束，之后的循环也不再执行
　　　continue：直接进行下一次循环，跳过当前循环

数组
　　　各个计算机语言都有的一个数据结构，最简单，最容易理解的数据结构
　　　数组，即一组数据

　　　定义一个数组，数组底层的本质是引用类型
　　　　int[] s = new int[5];
　　　　int[]其实就是类型名，int类型的数组，用int[]定义一个s，s就是一个整型的数组，类似于int s，只不过这里把int换成了int[]
　　　数组定义后就会有默认值
　　　　int，0
　　　　float，0.0
　　　　boolean，false
　　　　String，false
　　　定义的同时初始化
　　　　int[] s = new int[]{1,2,3,4,5};
　　　　int[] s = {1,2,3,4,5};

　　　数组的长度
　　　　s.length;

　　　查看数组
　　　Arrays.toString(s);//对数组s进行查看

随机数
　　　Math.random();
　　　范围[0,1)
　　　　　int[] p = new int[10];
　　　　　for(int i = 0; i < p.length; i++) {
　　　　　　　p[i] = (int)(Math.random() * 100);
　　　　　}
　　　　　System.out.println(Arrays.toString(p));

算法：选择排序
　　　int a[] = {3, 85, 49, 22, 30};
　　　for (int j = 0; j < a.length - 1; j++) {//总轮数，5个数排4轮
　　　　　for (int i = j + 1; i < a.length; i++) {//每轮都要找到一个最大值，进行交换
　　　　　　　if (a[j] < a[i]) {
　　　　　　　　　int temp = a[j];
　　　　　　　　　a[j] = a[i];
　　　　　　　　　a[i] = temp;
　　　　　　　}
　　　　　}
　　　}
　　　System.out.println(Arrays.toString(a));

算法：冒泡排序
　　　int[] a = new int[10];
　　　for (int i = 0; i < a.length; i++) {
　　　　　a[i] = (int)(Math.random() * 100);
　　　}
　　　System.out.println(Arrays.toString(a));
　　　for (int i = 0; i < a.length - 1; i++) {// 总轮数9轮
　　　　　for (int j = 0; j < a.length - i - 1; j++) {// 每找出一个最大值，下一轮就会少一个排序
　　　　　　　if (a[j] > a[j + 1]) {// 特别注意因为此时最后两个比较是用到了j和j+1，所以最后一个不再进行比较，到第length-1个结束
　　　　　　　　　int temp = a[j];
　　　　　　　　　a[j] = a[j + 1];
　　　　　　　　　a[j + 1] = temp;
　　　　　　　}
　　　　　}
　　　}
　　　System.out.println(Arrays.toString(a));

函数
　　　public static void getSum(int x,int[] y){
　　　　　函数体
　　　}
　　　getSum(10,y);//函数的调用，只有被调用，函数体才开始被执行

　　　public：
　　　　　公有的，还有privat私有的，protect受保护的
　　　static：
　　　　　静态的
　　　void：
　　　　　返回值类型，void（无返回值）、int、double、boolean等
　　　getSum：
　　　　　函数名：字母、数字、下划线，不能以数字开头，不能是关键字，可以是$。
　　　　　命名规范：驼峰式，getElementByMyName
　　　(int x,int[] y)：
　　　　　参数列表，形参x、y用于传参，把实参传递到函数体内部进行处理

　　　一个函数可以被反复调用

　　　函数可以定义在任意位置，与调用顺序无关

　　　一个函数可以调用另外一个函数

　　　函数的重载：
　　　　　1、函数名相同
　　　　　2、参数列表不同
　　　　　　　 1.参数的个数不同
　　　　　　　 2.参数的类型不一样
　　　　　　　 3.参数的顺序不同
　　　　　　　 4.和返回值无关 

package com.yuan;

    public static void main(String[] args) {
        for (int i = 0; i < 5; i++) {
            String star = new String("*");// String 是引用类型
            String star1 = "&";
        }
        
        // 打印五行五列 *
        for (int i = 0; i < 5; i++) {
            for (int j = 0; j < 5; j++) {
                System.out.print("*" + " ");
            }
            System.out.println();
        }

        for (int i = 0; i < 5; i++) {
            String star = "";
            for (int j = 0; j < 5; j++) {
                star = star + "*" + " ";
            }
            System.out.println(star);
        }

        // 打印等腰三角形
        int n = 4;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n - i; j++) {// 第i行前面有n-i个空格，后面的空格不用管
                System.out.print(" ");
            }
            for (int k = 1; k <= 2 * i - 1; k++) {// 按照1、3、5...打印，第i行有2*i-1个*
                System.out.print("*");
            }
            System.out.println();// 换行
        }

        int m = 4;
        for (int i = 1; i <= m; i++) {
            String line = "";
            for (int j = 1; j <= m - i; j++) {// 第i行前面有n-i个空格，后面的空格不用管
                line += " ";
            }
            for (int k = 1; k <= 2 * i - 1; k++) {// 按照1、3、5...，第i行有2*i-1个*
                line += "*";
            }
            System.out.println(line);// 打印加换行
        }
        
        // 打印九九乘法表
        for (int i = 1; i <= 9; i++) {
            for (int j = 1; j <= i; j++) {
                System.out.print(i + "*" + j + "=" + i * j + "\t");
            }
            System.out.println();
        }

        // 打印三位数的素数
        for (int i = 100; i < 1000; i++) {
            boolean temp = true;
            for (int j = 2; j < i; j++) {
                if (i % j == 0) {
                    temp = false;
                    break;
                }
            }
            if (temp) {
                System.out.println(i);
            }
        }

        String a = "absolute";// 字符串
        char b = 'b';// 字符
        System.out.println(a + b);

        s[1] = 98;

        s[3] = 89;

        System.out.println(Arrays.toString(s));// 查看数组Arrays.toString(s)

        double a2 = Math.random();
        System.out.println(a1 + "\n" + a2);

        for(int i = 0; i < p.length; i++) {
             p[i] = (int)(Math.random() * 100);
        }
        System.out.println(Arrays.toString(p));
        // 选择排序
        int a[] = {3, 85, 49, 22, 30};
        for (int j = 0; j < a.length - 1; j++) {// 总轮数，5个数排4轮
            for (int i = j + 1; i < a.length; i++) {// 每轮都要找到一个最大（小）值，进行交换
                if (a[j] < a[i]) {
                    int temp = a[j];
                    a[j] = a[i];
                    a[i] = temp;
                }
            }
        }

        System.out.println(Arrays.toString(a));

            b[i] = (int)(Math.random() * 100);
        }
        for (int i = 0; i < b.length - 1; i++) {
            for (int j = i + 1; j < b.length; j++) {
                if (b[j] < b[i]) {

                    b[i] = b[j];
                    b[j] = temp;
                }
            }
        }

        int[] a = new int[10];
        for (int i = 0; i < a.length; i++) {
            a[i] = (int)(Math.random() * 100);
        }
        System.out.println(Arrays.toString(a));
        for (int i = 0; i < a.length - 1; i++) {// 总轮数9轮
            for (int j = 0; j < a.length - i - 1; j++) {// 每找出一个最大值，下一轮就会少一个排序
                if (a[j] > a[j + 1]) {// 特别注意因为此时最后两个比较是用到了j和j+1，所以最后一个不再进行比较，到第length-1个结束

    public static void main(String[] args) {// main函数内调用函数

        System.out.println(sum(100));
        int t = random(50, 60);
        System.out.println(t);
        int u = random(4);
        System.out.println(u);
    }

    public static int sum(int n) {
        int s = 0;
        for (int i = 1; i <= n; i++) {
            s += i;
        }
        return s;
    }

    // 生成[m,n)之间的额随机整数
    public static int random(int m, int n) {
        int result = (int)(Math.random() * (n - m) + m);
        return result;
    }

    // 生成[0,n)之间的随机整数，
    // 重载了random函数，并且调用了函数random(m,n)
    public static int random(int n) {// 参数表不一样，数量不一样，使得两个同名函数random不一样
        return random(0, n);
    }
}
