MySQL数据库

2023/2/6 学习 MySQL

  • DDL（Data Definition Language）：数据定义语言。

    即数据定义语言，例如建数据库、建表等，都属于数据定义语言

    • 创建库，删除库
    • 创建视图
    • 函数
    • 存储过程
    • 触发器

  • DCL（Data Control Language）：数据控制语言。

    数据控制语言，如Grant、Rollback等等，常见于数据库安全管理，多数人一般很少用。

    • 创建用户，删除用户
    • 赋予权限
    • 删除权限

  • DML（Data Manipulation Language）：* 数据操作语言。

    即数据操纵语言，最常用的增删改查就属于DML，操作对象是数据表中的记录

    对表操作

    • insert
    • updata
    • delete

  • DQL（Data Query Language）：** 输出查询语言。

    对表操作

    • select
  • 只有分组列和聚合函数列，可以作为结果集的列
  • WHERE是定义在GROUP BY之前,先筛选,再分组(先where再GROUP BY)
  • HAVING ON 是分组之后的筛选条件,只能是分组列和聚合函数列
-- 三、结果排序 ORDER BY, ASC代表升序（可省略），DESC代表降序
SELECT * FROM t_student WHERE sex = '女' ORDER BY height DESC
-- --  多列排序
SELECT * FROM t_student WHERE sex = '女' ORDER BY height DESC, weight ASC

-- 四、限定数量查询,LIMIT 和 OFFSET。
-- 1个参数，表示省略第1个参数，表示查询出几行，
-- 2个参数，第1个参数表示第几行开始查，第2个参数表示查询出几行
SELECT * FROM t_student ORDER BY weight DESC LIMIT 5 -- 等同于LIMIT 0, 10
SELECT * FROM t_student ORDER BY weight DESC LIMIT 5, 10

SELECT * FROM t_student ORDER BY weight DESC LIMIT 10 OFFSET 5 -- 等同于LIMIT 5, 10

-- 五、分组查询。GROUP BY，聚合函数AVG(),SUM(),MAX(),MIN(),COUNT()
SELECT COUNT(*) FROM t_student
SELECT MAX(height) FROM t_student

-- 只有分组列和聚合函数列，可以作为结果集的列
SELECT sex, COUNT(*) cnt FROM t_student GROUP BY last_name ORDER BY cnt DESC LIMIT 3
SELECT sex, AVG(height) FROM t_student GROUP BY sex
SELECT sex, MIN(height) FROM t_student GROUP BY sex
SELECT sex, SUM(iq) FROM t_student GROUP BY sex

-- WHERE是定义在GROUP BY之前,先筛选,再分组(先where再GROUP BY)
-- HAVING ON 是分组之后的筛选条件,只能是分组列和聚合函数列
SELECT sex, COUNT(*) cnt 
FROM t_student 
GROUP BY last_name 
HAVING cnt > 4

-- 多列分组
SELECT sex, YEAR(birthday), COUNT(*) cnt 
FROM t_student 
GROUP BY sex, YEAR(birthday) 
HAVING cnt > 30

-- 六、去重 DISTINCT
SELECT DISTINCT sex FROM t_student
SELECT COUNT(DISTINCT sex) FROM t_student

CREATE FUNCTION f_days_of_year(p_year int) RETURNS INT
BEGIN
    IF (p_year % 4 = 0 AND p_year % 100 <> 0 OR p_year % 400 = 0)
        THEN RETURN 366;
    ELSE 
        RETURN 365;
    END IF;
END 

SELECT f_days_of_year(YEAR(birthday)) FROM t_employee

CREATE FUNCTION f_jiecheng(n INT) RETURNS BIGINT
BEGIN
    DECLARE result BIGINT DEFAULT 1;
    DECLARE i INT DEFAULT 1;
    WHILE i<n DO
        SET result = result * i;
        SET i = i + 1;
    END WHILE;
    RETURN result;
END

DROP FUNCTION f_jiecheng

CREATE FUNCTION f_get_level(id int) RETURNS char(2)
BEGIN
    DECLARE avg_iq FLOAT; 
    DECLARE current_iq INT;
    SELECT AVG(iq) INTO avg_iq FROM t_student t WHERE class_id = (
        SELECT class_id FROM t_student WHERE t.id = id
    );
    SELECT iq INTO current_iq FROM t_student t1 WHERE t1.id = id;
    IF current_iq > avg_iq 
        THEN RETURN '及格';
    ELSE 
        RETURN '淘汰';
    END IF;
END

SELECT f_get_level(id) FROM t_student

-- 存储过程，一段代码，没有returns，批量执行
CREATE PROCEDURE p_delete_student(id INT)
BEGIN
    DECLARE a char(2);-- 是否及格
    DECLARE id INT;
    DECLARE con INT DEFAULT 0;-- 用于判断是否退出游标循环的标志
    DECLARE cur CURSOR FOR SELECT id FROM t_student;-- 游标
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET con = 1;
    
    OPEN cur;
    FETCH cur INTO id;
    WHILE con = 0 DO
        FETCH cur INTO id;
        SELECT f_get_level(id) INTO a FROM t_student;
        IF a = '淘汰' THEN
            DELETE FROM t_student t WHERE t.id = id;
        END IF;
        FETCH cur INTO id;
    END WHILE;

    
    CLOSE cur;
    
END

call p_delete_student
