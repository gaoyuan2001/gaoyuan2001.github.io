/**
 * @Title: JdbcUtils
 * @Project: weekhomework0211
 * @Description: 工具类，供Dao层调用
 * @Author: 14508
 * @Version:

public class JdbcUtils {
    private JdbcUtils() {
    }

    public static Connection getConnection(String driver, String url, String user, String password) {
        try {
            Class.forName(driver);
            return DriverManager.getConnection(url, user, password);
        } catch (Exception e) {
            e.printStackTrace();
            throw new RuntimeException("无法获取数据库连接", e);
        }
    }

    public static void closeConnection(Connection conn) {
        if (conn != null) {
            try {
                conn.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }

    public static void closeStatement(Statement stmt) {
        if (stmt != null) {
            try {
                stmt.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }

    public static void closeResultSet(ResultSet rs) {
        if (rs != null) {
            try {
                rs.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }

    public static void close(Object... args) {
        for (Object obj : args) {
            if (obj instanceof Connection conn) {
                closeConnection(conn);
            }
            if (obj instanceof Statement stmt) {
                closeStatement(stmt);
            }
            if (obj instanceof ResultSet rs) {
                closeResultSet(rs);
            }
        }
    }

    public static ResultSet query(Connection conn, String sql, List<Object> args) {
        try {
            PreparedStatement ps = conn.prepareStatement(sql);
            for (int i = 0; i < args.size(); i++) {
                ps.setObject(i + 1, args.get(i));
            }
            return ps.executeQuery();
        } catch (Exception e) {
            e.printStackTrace();
        }
        return null;
    }

    public static ResultSet query(Connection conn, String sql, Object... args) {
        return query(conn, sql, Arrays.asList(args));
    }

    public static <T> T query(Connection conn, String sql, ResultSetHandler<T> handler, List<Object> args) {
        ResultSet rs = query(conn, sql, args);
        return handler.handle(rs);
    }

    public static <T> T query(Connection conn, String sql, ResultSetHandler<T> handler, Object... args) {
        return query(conn, sql, handler, Arrays.asList(args));
    }


    public interface ResultSetHandler<T> {
        T handle(ResultSet rs);
    }

    public static class BeanListResultSetHandler<T> implements ResultSetHandler<List<T>> {
        private Constructor<T> constructor;
        private Class<T> clazz;
        private Map<String, PropertyDescriptor> pdMap = new HashMap<>();


        public BeanListResultSetHandler(Class<T> clazz) {
            this.clazz = clazz;

            try {
                this.constructor = clazz.getDeclaredConstructor();
                BeanInfo bi = Introspector.getBeanInfo(clazz);
                //PropertyDescriptor描述了Java Bean通过一对存取方法导出的一个属性。
                PropertyDescriptor[] pds = bi.getPropertyDescriptors();
                for (PropertyDescriptor pd : pds) {
                    pdMap.put(pd.getName(), pd);
                }
            } catch (Exception e) {
                e.printStackTrace();
            }
        }

        @Override
        public List<T> handle(ResultSet rs) {
            List<String> columnLabels = new ArrayList<>();
            List<T> result = new ArrayList<>();

            try {
                ResultSetMetaData rsmd = rs.getMetaData();
                int cnt = rsmd.getColumnCount();

                for (int i = 0; i < cnt; i++) {
                    String columnLabel = rsmd.getColumnLabel(i + 1);
                    columnLabels.add(columnLabel);
                }

                while (rs.next()) {
                    T t = constructor.newInstance();
                    for (String column : columnLabels) {
                        Object obj = rs.getObject(column);

                        String property = underlineToCamel(column);
                        PropertyDescriptor pd = pdMap.get(property);

                        if (pd != null) {
                            Method m = pd.getWriteMethod();
                            m.invoke(t, convertType(obj, pd));
                        }
                    }
                    result.add(t);
                }
            } catch (Exception e) {
                e.printStackTrace();
            }

            return result;
        }
    }

    private static String underlineToCamel(String str) {
        if (str.contains("_")) {
            String[] tokens = str.split("_");
            StringBuilder result = new StringBuilder(tokens[0]);
            for (int i = 1; i < tokens.length; i++) {
                String firstLetter = String.valueOf(tokens[i].charAt(0));
                String remain = tokens[i].substring(1);
                result.append(firstLetter.toUpperCase()).append(remain);
            }
            return result.toString();
        } else {
            return str;
        }
    }

    private static Object convertType(Object obj, PropertyDescriptor pd) {
        if (obj == null) {
            return null;
        }
        Class<?> clazz = pd.getPropertyType();

        if (clazz == Integer.class) {
            return ((Number) obj).intValue();
        } else if (clazz == Short.class) {
            return ((Number) obj).shortValue();
        } else if (clazz == Byte.class) {
            return ((Number) obj).byteValue();
        } else if (clazz == Long.class) {
            return ((Number) obj).longValue();
        } else if (clazz == Float.class) {
            return ((Number) obj).floatValue();
        } else if (clazz == Double.class) {
            return ((Number) obj).doubleValue();
        } else if (clazz == LocalTime.class || clazz == LocalDate.class || clazz == LocalDateTime.class) {
            return convertLocalDateTime(((Date) obj).getTime(), clazz);
        }
        return obj;
    }

    private static Object convertLocalDateTime(long time, Class<?> clazz) {
        Instant is = Instant.ofEpochMilli(time);
        LocalDateTime ldt = LocalDateTime.ofInstant(is, ZoneId.systemDefault());
        if (clazz == LocalDate.class) {
            return ldt.toLocalDate();
        } else if (clazz == LocalTime.class) {
            return ldt.toLocalTime();
        } else {
            return ldt;
        }
    }

    public static int update(Connection conn, String sql, List<Object> args) {
        try {
            PreparedStatement ps = conn.prepareStatement(sql);
            for (int i = 0; i < args.size(); i++) {
                ps.setObject(i + 1, args.get(i));
            }
            return ps.executeUpdate();
        } catch (Exception e) {
            e.printStackTrace();
            return 0;
        }
    }
}

        Connection conn = null;
        try {
            conn = Global.getConnection();
            String sql = "select * from t_employee where emp_id = ?";

            List<Employee> employees = JdbcUtils.query(conn, sql, new JdbcUtils.BeanListResultSetHandler<>(Employee.class), empId);
            return employees.size() == 0 ? null : employees.get(0);
        } finally {
            JdbcUtils.close(conn);
        }
    }
