冒泡排序：也叫作交换排序

    /**
     * 冒泡排序（交换排序）
     */
    public static void bubbleSort(int[] n) {
        for (int i = 0; i < n.length - 1; i++) {
            for (int j = 0; j < n.length - 1 - i; j++) {
                if (n[j + 1] > n[j]) {
                    int t = n[j + 1];
                    n[j + 1] = n[j];
                    n[j] = t;
                }
            }
        }
    }

选择排序

    /**
     * 选择排序
     */
    public static void selectSort(int[] n) {
        for (int i = 0; i < n.length - 1; i++) {
            for (int j = i + 1; j < n.length; j++) {
                if (n[i] > n[j]) {
                    int t = n[i];
                    n[i] = n[j];
                    n[j] = t;
                }
            }
        }
    }

插入排序：其中有两种变体

    /**
     * 插入排序
     */

    /**
     * 找到要插入的点，将后续整体移动
     * @param n
     */
    public static void insertSort1(int[] n) {
        for (int i = 1; i < n.length; i++) {
            for (int j = 0; j < i; j++) {
                if (n[i] > n[j] && n[i] < n[j + 1]) {
                    int t = n[i];
                    for (int k = i - 1; k >= j + 1; k++) {
                        n[k + 1] = n[k];
                    }
                    n[j + 1] = t;
                }
            }
        }
    }

    /**
     * 每向前走一次，后移一位，当找到要插入的点，直接插入
     * @param n
     */
    public static void insertSort2(int[] n) {
        for (int i = 1; i < n.length - 1; i++) {//从1开始选择需要插入的元素，小于0的位置，则把0处后移，然后赋值于0处
            int t = n[i];
            int j = i;
            while (j > 0 && t < n[j - 1]) {//需要从0开始判断插入位置
                n[j] = n[j - 1];
                j--;
            }
            if (j != i) {
                n[j] = t;
            }
        }
    }

快速排序：也叫做快排，重点看，一定要理解

    /**
     * 快速排序
     */
    public static void quickSort(int[] n) {
        quickSortInner(n, 0, n.length - 1);
    }

    /**
     * 嵌套的思想
     * 思路：选择出来基准后，先保存下来，此时基准处的值可以被覆盖
     * 当选择的基准值是表头的位置时，必须从后指针开始判断，因为此时前指针所指的元素可以被赋值（当然如果选择基准值为表尾的位置时，则相反）
     * 当前指针所指被后指针处的元素赋值后，表示前指针的元素已经确定，此时需要把前指针向后移一位，以求前指针所指一直为需要排序的元素
     * 因为后指针所指的值已经确定位置，所以后指针所指的位置又可以被前指针所赋值
     * 循环往复，直到每一个小组的元素个数为1时，也就是数组排好序了
     * （通俗来说，刚开始看成基准值的位置为空，赋值后，不再为空，指针后移，已经把值赋走的位置又可以被看为空，又可以继续被赋值，循环往复）
     *
     * @param n
     * @param left
     * @param right
     */
    public static void quickSortInner(int[] n, int left, int right) {
        //结束的标志：当每一个小小小小组的元素为1时，代表所有的元素已经排好序，因为当数组元素个数为1时，自动判断此数组有序
        if (left >= right) {
            return;
        }
        int i = left;
        int j = right;

        int pivot = n[left];

        while (i != j) {
            while (n[j] >= pivot && j > i) {//当后指针处的值一直比基准值pivot要大时，则不需要改变其位置，而是把后指针指向前一个元素
                j--;
            }
            //这里跳出循环，意味着发现指针所指的元素要比pivot小，需要改变位置（不是交换位置）
            if (i < j) {
                n[i] = n[j];//一定要注意这里不是交换两者的值，而只是后指针赋值给前指针
                i++;//这里是i已经被赋完值了，值已经确定了，这一轮已经不再需要再进行排序改动了，前指针自然要移动到后一个需要排序的元素身上
            }
            while (n[i] <= pivot && i < j) {//当前指针处的值一直比基准值pivot要小时，则不需要改变其位置，而是把前指针指向后一个元素
                i++;
            }
            //这里跳出循环，意味着发现指针所指的元素要比pivot大，需要改变位置（不是交换位置）
            if (i < j) {
                n[j] = n[i];//这里一样要注意，不再是交换两者的值，而只是前指针赋值给后指针
                j--;//后指针j处已经被赋完值，这一轮已经不需要再排序改动了，后指针自然要移动到前一个需要排序的元素身上
            }
        }
        n[i] = pivot;
        //System.out.println(Arrays.toString(n));
        quickSortInner(n, left, i - 1);//这里容易出错，记得不要用 0和n.length
        quickSortInner(n, i + 1, right);
    }
