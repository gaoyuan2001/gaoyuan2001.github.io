js的事件处理：交互的基础

定义事件的三种方式：

1.直接写在行内。写成onXXX的形式。

  • 缺陷：js代码长度有限，且阅读、调试不方便，

  • 解决方案可以调用函数，而函数体写在JavaScript中

  • 同一类型的事件，只能定义（生效）一个

2.dom.onXXX

  • 本质即修改标记的哦onXXX属性，也就是说，同一种方式一致

  • 可以取消事件：dom.onXXX = null;

3.dom.addEventListener(“xxx”,fun);推荐的方式（事件监听）

  • 第一个参数：事件类型，第二个参数：事件处理函数（函数名，或者匿名函数体）

  • 可以对同一类型的事件添加多次（按定义的顺序调用）

  • 可以取消事件：dom.removeEventListener(“xxx”,fun);移除事件处理函数，

    第一个参数：事件类型，第二个参数：事件处理函数（函数名），匿名函数体无法移除

事件对象：封装了事件触发现场的一些相关数据

1.内置的事件对象：event

2.在事件处理函数中，第一个参数即为event对象

onXXX = “js代码”

常用事件的类型：

1.onclick=”js代码”，单击事件，鼠标或者回车都可以，

2.ondbclick，双击事件

3.键盘相关事件：keydown，键盘按下，keyup，键盘弹起，keypressed

keydown可以一直触发，keyup只触发一次

4.鼠标相关事件：mousedown，mouseup，mouseleave，mouseover，mouseenter，mousemove

5.载入事件：window.onload = function( ){……};//当窗口加载完毕之后执行，里面可以放js

dom属性：

1.offsetTop：上偏移

2.offsetLeft：左偏移

3.offsetParent：绝对定位的参照物dom

引入js的三种方式

1.行内js，只能用于事件处理属性。onXXX

2.页内js：

3.引入外部js：script src=””

事件委托（基于事件冒泡）

event.target：用于获取事件源，触发事件的对象

使用场景：

1.大量相同（或相似）的元素添加事件监听器，会导致事件监听器过多。

2.预期有不存在的元素，提前添加事件（设置在父元素上），可以用target来获取触发的元素是不是原本要设置的元素

<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <title></title>
        <style type="text/css">
            .outer {
                width: 400px;

                background-color: pink;
            }

            .inner {
                width: 200px;
                height: 100px;
                background-color: gold;
            }
        </style>
        <script type="text/javascript">
            window.onload = function() {
                let inner = document.querySelector(".outer");
                /* for (let i = 0; i < inner.length; i++) {
                    let a = inner[i];
                    a.addEventListener("click", function(e) {
                        console.log("aaaa");
                    });
                } */
                inner.addEventListener("click",function(e){
                    if(e.target.classList.contains("inner")){
                        console.log("aaaa");
                    }
                });

                let btn = document.querySelector("#btn");
                let outer = document.querySelector(".outer");
                btn.addEventListener("click", function() {
                    let i = document.createElement("div");
                    i.className = "inner";
                    outer.appendChild(i);
                });
            };
        </script>
    </head>
    <body>
        <div class="outer">
            <div class="inner"></div>

        </div>

动画：随时间有节奏的变化

  • setTimeout(… , …)：定时器（函数行为，时间）

  • setInterval(… , …)：循环定时器

  • clearTimeout(… , …)：清除定时器

  • clearInterval(… , …)：清除循环定时器

<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <title></title>
        <style type="text/css">
            html,
            body {
                height: 100%;
            }

            body {
                margin: 0;
            }

            .my {
                width: 80px;
                height: 80px;
                background-color: pink;
                position: absolute;
                top: 50%;
                left: 50%;
            }

            #btn {
                position: absolute;
                bottom: 5px;
                right: 5px;
            }
        </style>
        <script type="text/javascript">
            window.onload = function() {
                let my = document.querySelector(".my");
                let btn = document.querySelector("#btn");
                btn.addEventListener("click", function() {
                    //创建定时器
                    /* let s1 = setTimeout(function() {
                        my.remove();
                    }, 3000); */

                    //清除定时器
                    /* clearTimeout(s1);

                    let s2 = setInterval(function() {
                        console.log("aaa");
                    }, 1000); */

                    /* clearInterval(s2); */


                    let duration = parseInt(Math.random() * 600);

                    let timer;

                    setInterval(function() {
                        if (timer) {
                            clearInterval(timer);
                        }
                        
                        let direction = parseInt(Math.random() * 4);
                        let speed = parseInt(Math.random() * 20 + 20);
                        timer = move(my, direction, speed);
                    }, duration);

                });
            };

            function move(dom, direction, speed) {
                //0表示上，1表示下，2表示左，3表示右
                switch (direction) {
                    case 0: {
                        return setInterval(function() {
                            dom.style.top = dom.offsetTop - 5 + "px";
                        }, speed);
                        break;
                    }
                    case 1: {
                        return setInterval(function() {
                            dom.style.top = dom.offsetTop + 5 + "px";
                        }, speed);
                        break;
                    }
                    case 2: {
                        return setInterval(function() {
                            dom.style.left = dom.offsetLeft - 5 + "px";
                        }, speed);
                        break;
                    }
                    case 3: {
                        return setInterval(function() {
                            dom.style.left = dom.offsetLeft + 5 + "px";
                        }, speed);
                        break;
                    }
                }
            }
        </script>
    </head>
    <body>
        <div class="my"></div>
        <button id="btn">点我</button>
        
        <a href="demo5.html">去下一页</a>
    </body>
</html>

bom：Browser Object Mode，浏览器对象模型

通过js可以控制浏览器中的一些对象

window：window.document

location：指代地址栏对象

  • location.href=””;//给href赋新的属性，载入新页面
  • location.reload();//刷新
  • location.replace();

history：指代访问历史记录

  • history.back();//后退一个页面
  • history.forward();//前进一个界面
  • history.go(-5);//后退5的界面
<!DOCTYPE html>
<html>

        <title></title>
    </head>
    <body>

        <script type="text/javascript">
            let btn = document.querySelector("#btn");
            btn.addEventListener("click",function(){
                //location.href = "http://www.baidu.com";
                //location.reload();//刷新
                //location.replace("demo1.html");
                
                history.back();//后退一步
                history.forward();//前进一步
                
                history.go(-5);//后退5步
                history.go(5);//前进5步	
            });
        </script>
    </body>
</html>
